					value = 30

				add_dip_power = 25

				add_splendor = 200
			}
		}
		if = {
			limit = { exists = SCA }

				add_treasury = 1000
				add_years_of_income = 1
				add_adm_power = 25
				add_dip_power = 25
				add_mil_power = 20
				add_splendor = 200
			}
		}
		every_owned_province = {
			limit = {
				OR = {
					controlled_by = SWE
					controlled_by = SCA
				}
			}
			add_devastation = 60
			add_base_tax = -1
			add_base_production = -1
		}
		every_owned_province = {
			limit = {
				OR = {
					NOT = { controlled_by = SWE }
					NOT = { controlled_by = SCA }
				}
			}
			add_devastation = 20
		}
	}

country_event = { # Chmielnicki's Uprising
	id = vu_plc.2
	title = vu_plc.2.t
	desc = vu_plc.2.d
	picture = SIEGE_eventPicture
	
	major = yes
	
	fire_only_once = yes
	
	trigger = {
		is_year = 1600
		OR = {
			tag = POL
			tag = PLC
		}
		estate_influence = {
			estate = estate_cossacks
			influence = 20
		}
		NOT = {
			estate_loyalty = {
				estate = cossacks
				loyalty = 55
			}
		}
		any_owned_province = {
			OR = {
				area = zaporizhia_area
				area = yedisan_area

				area = sloboda_ukraine_area
				area = lower_dnieper_area
			}
			is_core = ROOT
			is_core = ZAZ
		}
		NOT = { exists = ZAZ }
	}
	
	mean_time_to_happen = {
		months = 12
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = {
					2407 = {
						owned_by = ROOT
					}
				}

					4542 = {
						owned_by = ROOT
					}
				}

					add_core = ZAZ
				}
			}
			if = {
				limit = {
					7255 = {
						owned_by = ROOT
					}
				}
				7255 = {
					add_core = ZAZ
				}
			}
			if = {
				limit = {
					7256 = {
						owned_by = ROOT
					}
				}
				7256 = {
					add_core = ZAZ
				}
			}
			release = ZAZ
			ZAZ = {
				capital_scope = {
					build_to_forcelimit = {
						infantry = 0.5
						cavalry = 0.3
						artillery = 0.2
					}
				}
				capital_scope = {
      					infantry = ZAZ
      					infantry = ZAZ
      					infantry = ZAZ
      					infantry = ZAZ
      					infantry = ZAZ

					shock = 5
					fire = 4
					manuever = 5
					siege = 4
					name = "Bohdan Chmielnicki"
				}
			}
		}
	}

		name = vu_plc.2.a
		ai_chance = { factor = 25 }
		every_owned_province = {
			limit = {
				OR = {
					area = west_dniepr_area
					area = eastern_volhynia_area
					area = east_dniepr_area

					province_id = 1943
					province_id = 7206
					province_id = 3241
					province_id = 4540
				}
			}
			cede_province = ZAZ
		}

		create_march = ZAZ

			random_owned_province = {
				limit = {
					is_core = PLT
				}
				spawn_rebels = {
					type = nationalist_rebels
					size = 1.5
					friend = PLT
				}
			}
		}
	}
	
	option = { # Chmielnicki's uprising starts
		name = vu_plc.2.b
		ai_chance = { factor = 75 }
		set_global_flag = chmielnicki_uprising
		ZAZ = {

				}

					type = cavalry
					amount = 5
					speed = 1
					cost = 0
				}
			}
		}
		add_stability = -1
	}
}
